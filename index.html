Tìm Tổng Lớn Nhất K x K

#include <iostream>
#include <vector>
using namespace std;

int main() {
    int n, k;
    cin >> n >> k;
    vector<vector<int>> matrix(n, vector<int>(n));

    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < n; ++j) {
            cin >> matrix[i][j];
        }
    }

    int maxSum = 0;

    for (int i = 0; i <= n - k; ++i) {
        for (int j = 0; j <= n - k; ++j) {
            int currentSum = 0;
            for (int x = 0; x < k; ++x) {
                for (int y = 0; y < k; ++y) {
                    currentSum += matrix[i + x][j + y];
                }
            }
            maxSum = max(maxSum, currentSum);
        }
    }

    cout << "Tổng Lớn Nhất: " << maxSum << endl;
    return 0;
}
        
using System;
using System.Linq;

class Program {
    static void Main() {
        int n = int.Parse(Console.ReadLine());
        int k = int.Parse(Console.ReadLine());
        int[,] matrix = new int[n, n];

        for (int i = 0; i < n; i++) {
            var values = Console.ReadLine().Split(',').Select(int.Parse).ToArray();
            for (int j = 0; j < n; j++) {
                matrix[i, j] = values[j];
            }
        }

        int maxSum = 0;

        for (int i = 0; i <= n - k; i++) {
            for (int j = 0; j <= n - k; j++) {
                int currentSum = 0;
                for (int x = 0; x < k; x++) {
                    for (int y = 0; y < k; y++) {
                        currentSum += matrix[i + x, j + y];
                    }
                }
                if (currentSum > maxSum) {
                    maxSum = currentSum;
                }
            }
        }

        Console.WriteLine("Tổng Lớn Nhất: " + maxSum);
    }
}
        
n, k = map(int, input().split())
matrix = [list(map(int, input().split(','))) for _ in range(n)]

max_sum = 0

for i in range(n - k + 1):
    for j in range(n - k + 1):
        current_sum = 0
        for x in range(k):
            for y in range(k):
                current_sum += matrix[i + x][j + y]
        max_sum = max(max_sum, current_sum)

print("Tổng Lớn Nhất:", max_sum)
